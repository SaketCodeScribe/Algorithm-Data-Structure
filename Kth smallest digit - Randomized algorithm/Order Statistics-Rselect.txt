#include <bits/stdc++.h>
using namespace std;

int partition(vector<int> &arr,int low,int high,int pivot){
    int temp,j,i;
/*    cout<<pivot<<" "<<arr[pivot]<<endl;
    for (int k=low;k<=high;k++)
    cout<<arr[k]<<" ";
    cout<<endl;*/
    i=low;
    temp=arr[pivot];
    arr[pivot]=arr[low];
    arr[low]=temp;
    for(j=low+1;j<=high;){
        if(arr[j]>=arr[low])
        j++;
        else {
            i++;
            temp=arr[i];
            arr[i]=arr[j];
            arr[j]=temp;
            j++;
        }
    }
    temp=arr[low];
    arr[low]=arr[i];
    arr[i]=temp;
    /*for (int k=low;k<=high;k++)
    cout<<arr[k]<<" ";
    cout<<endl<<"i= "<<i<<endl;*/
    return i;
}

int Rselect(vector<int> &arr,int low,int high,int k){
    int pivot;
    srand(time(NULL));
    if(low<high)
    pivot=low+rand()%(high-low);
    else pivot=low;
    int p=partition(arr,low,high,pivot);
    //cout<<"p="<<p<<endl;
    if(p==k)
    return k;
    else if(p<k)
        Rselect(arr,p+1,high,k);
    else Rselect(arr,low,p-1,k);
}

int main() {
	int n,i,x,k;
	vector<int> arr;
	cin>>n>>k;
	for(i=0;i<n;i++){
	    cin>>x;
	    arr.push_back(x);
	}
	if(k>n||k<=0)
	cout<<"Wrong input"<<endl;
	else cout<<arr[Rselect(arr,0,n-1,k-1)]<<endl;
	return 0;
}
