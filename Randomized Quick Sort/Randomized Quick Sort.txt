#include <bits/stdc++.h>
using namespace std;

int partition(vector<int> &arr,int low,int high,int pivot){
    int temp,j,i;
/*    cout<<pivot<<" "<<arr[pivot]<<endl;
    for (int k=low;k<=high;k++)
    cout<<arr[k]<<" ";
    cout<<endl;*/
    i=low;
    temp=arr[pivot];
    arr[pivot]=arr[low];
    arr[low]=temp;
    for(j=low+1;j<=high;){
        if(arr[j]>=arr[low])
        j++;
        else {
            i++;
            temp=arr[i];
            arr[i]=arr[j];
            arr[j]=temp;
            j++;
        }
    }
    temp=arr[low];
    arr[low]=arr[i];
    arr[i]=temp;
    /*for (int k=low;k<=high;k++)
    cout<<arr[k]<<" ";
    cout<<endl<<"i= "<<i<<endl;*/
    return i;
}

void RQuicksort(vector<int> &arr,int low,int high){
    int pivot;
    srand(time(NULL));
    if(low<high){
        pivot=low+rand()%(high-low);
        int p=partition(arr,low,high,pivot);
        RQuicksort(arr,p+1,high);
        RQuicksort(arr,low,p-1);
    }
}

int main() {
	int n,i,x;
